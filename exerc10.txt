#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main(){

    int matrizA[4][2], matrizB[4][2], matrizC[4][2];
    srand(time(NULL));
    
    for(int i=0; i < 4; i++){
        for(int j=0; j < 2; j++){
           matrizA[i][j]=(rand()%1000);
           matrizB[i][j]=(rand()%1000);
           matrizC[i][j]=matrizA[i][j]+matrizB[i][j];
           printf("[%d ] ", matrizC[i][j]);
        }
        printf("\n");
    }
